핑퐁 ④: 집합(Set) — 연산

Ping ④ 교수자 시연

class_A = {"수학", "영어", "과학", "체육"} class_B = {"영어", "과학", "음악", "미술"} print("공통 과목:", class_A & class_B) print("A만의 과목:", class_A - class_B) print("전체 과목:", class_A | class_B)

Pong ④ 학생 작성 — 확진자 급증 vs 감소 국가 비교

surge_countries = {"한국","미국","영국","인도","브라질"} decline_countries = {"일본","독일","미국","영국","호주"} both = ___ # 1) 교집합 (급증 & 감소 동시) surge_only = ___ # 2) 차집합 (급증만) all_countries= ___ # 3) 합집합 (전체) exclusive = ___ # 4) 대칭차집합 (한쪽만) print("급증 & 감소:", both) print("급증만:", surge_only) print("전체:", all_countries) print("한쪽만:", exclusive)
▶ 정답 확인 
both          = surge_countries & decline_countries
surge_only    = surge_countries - decline_countries
all_countries = surge_countries | decline_countries
exclusive     = surge_countries ^ decline_countries

도전 ②: 7일 이동평균(Moving Average) 직접 계산

이동평균이란 연속된 n일의 평균을 순차적으로 계산하는 것이다. 데이터의 단기 변동을 부드럽게 만들어 추세를 파악한다.

결과 리스트의 길이 = 원본 길이 - (윈도우 크기 - 1) = 21 - 6 = 15
cases_21 = [523,681,445,892,1023,756,634, 812,567,934,1105,678,489,723, 945,1067,834,612,778,1023,889] moving_avg = [] for i in range(len(cases_21) - 6): window = ___ # i번째부터 7개 슬라이싱 avg = ___ # 평균 계산 moving_avg.append(round(avg, 1)) print("7일 이동평균:", moving_avg) print("이동평균 길이:", len(moving_avg)) # 반드시 15 max_avg = ___ max_idx = ___ print(f"이동평균 최고: {max_avg} (시작일: {max_idx+1}일차)")
▶ 정답 확인 
window  = cases_21[i:i+7]
avg     = sum(window) / 7
max_avg = max(moving_avg)
max_idx = moving_avg.index(max_avg)
