```python
        total += numbers[i]

print(f"짝수 합: {total}")

print(f"짝수 합(컴프리헨션): {even_sum}")
```

**힌트:** `range` 범위, `continue` 뒤 코드 실행 여부, 짝수/홀수 조건을 확인하라.

### 디버그 ③: 함수 오류

```python
# AI에게 요청: "이동평균을 계산하는 함수를 짜줘"
# AI가 생성한 코드 (4곳의 오류가 있다)

def moving_average(data, window):
    result = []
    for i in range(len(data)):                      # 오류 1: 범위 초과
        segment = data[i:i+window]
        avg = sum(segment) / window
        result.append(avg)

ma = moving_average(daily, 3)
print(f"이동평균: {ma}")                             # 오류 2: 반환값 없음

# 결과 확인
print(f"원본 길이: {len(daily)}")
print(f"이동평균 길이: {len(ma)}")                    # 오류 3: ma는 None
```

### 해답 7. 함수 정의

```python
# (a) 최대공약수
def gcd(a, b):
    while b != 0:
        a, b = b, a % b
    return a

print(gcd(12, 8))   # 4

# (b) 중앙값
def median(data):
    s = sorted(data)
    n = len(s)
    if n % 2 == 1:
        return s[n // 2]
    else:
        return (s[n // 2 - 1] + s[n // 2]) / 2

print(median([3, 1, 4, 1, 5]))   # 3
print(median([3, 1, 4, 5]))      # 3.5

# (c) 분산
def variance(data):
    avg = sum(data) / len(data)
    return sum((x - avg) ** 2 for x in data) / len(data)

print(variance([2, 4, 6, 8, 10]))   # 8.0
```

### 해답 10. 코로나19 주간 리포트 생성기

```python
covid_30days = [
    523, 681, 445, 892, 1023, 756, 634,
    812, 567, 934, 1105, 678, 489, 723,
    945, 1067, 834, 612, 778, 1023, 889,
    1234, 567, 890, 1345, 1023, 678, 934,
    1156, 823
]

def weekly_report(data, week_num):
    start_idx = (week_num - 1) * 7
    end_idx = min(start_idx + 7, len(data))
    week_data = data[start_idx:end_idx]

    total = sum(week_data)
    avg = total / len(week_data)
    max_val = max(week_data)
    min_val = min(week_data)
    max_day = week_data.index(max_val) + 1
    min_day = week_data.index(min_val) + 1
    danger_days = len([x for x in week_data if x >= 1000])

    print(f"=== {week_num}주차 리포트 ({start_idx+1}일~{end_idx}일) ===")
    print(f"  총 확진자: {total:,}명")
    print(f"  일 평균: {avg:,.0f}명")
    print(f"  일 최다: {max_val:,}명 ({max_day}일차)")
    print(f"  일 최소: {min_val:,}명 ({min_day}일차)")

    if week_num > 1:
        prev_start = (week_num - 2) * 7
        prev_end = prev_start + 7
        prev_total = sum(data[prev_start:prev_end])
        change = (total - prev_total) / prev_total * 100
        arrow = "▲" if change > 0 else "▼"
        print(f"  전주 대비: {arrow} {abs(change):.1f}% {'증가' if change > 0 else '감소'}")
    else:
        print(f"  전주 대비: (첫 주)")

    print(f"  위험일(1000+): {danger_days}일")

for w in range(1, 5):
    weekly_report(covid_30days, w)
    print()
```

### 해답 11. (보너스) *args 활용

```python
def common_elements(*lists):
    if not lists:
        return set()
    result = set(lists[0])
    for lst in lists[1:]:
        result &= set(lst)
    return result
```

### 해답 12. (보너스) 간이 성적 관리 시스템

```python
students = {}

def add_student():
    name = input("이름: ")
    try:
        kor = int(input("국어 점수: "))
        eng = int(input("영어 점수: "))
        math = int(input("수학 점수: "))
        students[name] = {"국어": kor, "영어": eng, "수학": math}
        print(f"'{name}' 추가 완료.")
    except ValueError:
        print("점수는 숫자만 입력하세요!")

def print_report():
    if not students:
        print("등록된 학생이 없습니다.")
        return
    ranked = sorted(students.items(),
                    key=lambda x: sum(x[1].values()) / 3, reverse=True)
    print(f"{'이름':>6} {'국어':>4} {'영어':>4} {'수학':>4} {'평균':>6}")
    print("-" * 30)
    for name, scores in ranked:
        avg = sum(scores.values()) / 3
        print(f"{name:>6} {scores['국어']:>4} {scores['영어']:>4} "
              f"{scores['수학']:>4} {avg:>6.1f}")

def top_by_subject():
    if not students:
        print("등록된 학생이 없습니다.")
        return
    for subj in ["국어", "영어", "수학"]:
        top = max(students.items(), key=lambda x: x[1][subj])
        print(f"  {subj} 최고: {top[0]} ({top[1][subj]}점)")

def search_student():
    name = input("검색할 이름: ")
    if name in students:
        s = students[name]
        avg = sum(s.values()) / 3
        print(f"  {name}: 국어 {s['국어']}, 영어 {s['영어']}, 수학 {s['수학']} (평균: {avg:.1f})")
    else:
        print("찾을 수 없습니다.")

while True:
    print("\n=== 성적 관리 시스템 ===")
    print("1. 학생 추가  2. 성적표  3. 과목별 최고  4. 검색  5. 종료")
    choice = input("선택: ")
    if choice == "1":
        add_student()
    elif choice == "2":
        print_report()
    elif choice == "3":
        top_by_subject()
    elif choice == "4":
        search_student()
    elif choice == "5":
        print("종료합니다.")
        break
    else:
        print("1~5 중 선택하세요.")
```

```python
def moving_average(data, window):
    result = []
    for i in range(len(data) - window + 1):     # 수정 1: 범위 제한
        segment = data[i:i+window]
        avg = sum(segment) / window
        result.append(avg)
    return result                                  # 수정 2: return 추가

daily = [100, 200, 300, 400, 500, 600, 700]
ma = moving_average(daily, 3)
print(f"이동평균: {ma}")
```

## 3주차 핵심 정리

| 개념 | 문법 | 핵심 포인트 |
|---|---|---|
| **조건문**(if) | `if/elif/else:` | 큰 조건부터 검사, 콜론과 들여쓰기 필수 |
| **for문** | `for x in 시퀀스:` | `enumerate`, `zip`, `break`, `continue` |
| **while문** | `while 조건:` | 무한루프 주의, 종료 조건 필수 |
| **컴프리헨션** | `[식 for x in S if 조건]` | for문의 한 줄 압축, 딕셔너리도 가능 |
| **함수** | `def 이름(매개변수):` | `return`, `lambda`, `*args`, `**kwargs` |
| **예외처리** | `try/except/else/finally` | 구체적 예외를 잡고, 프로그램을 보호 |
